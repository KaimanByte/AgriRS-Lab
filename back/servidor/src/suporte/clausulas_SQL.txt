DROP TABLE IF EXISTS tb_adm, tbnoticias, tbpublicacao, tboportunidades;

CREATE TABLE tb_adm (
    id_adm INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    login VARCHAR(50) NOT NULL UNIQUE,
    senha VARCHAR(100) NOT NULL
);

CREATE TABLE tbnoticias (
    idnoticia INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    titulo TEXT NOT NULL,
    link TEXT,
    data DATE NOT NULL,
    exibir BOOLEAN DEFAULT TRUE,
    descricao TEXT,
    url_leiamais TEXT,
    imagem TEXT
);

CREATE TABLE tbpublicacao (
  idpublicacao SERIAL PRIMARY KEY,
  titulo VARCHAR(150) NOT NULL,
  descricao TEXT,
  imagem_url TEXT,
  pdf_url TEXT,
  citacao_url TEXT,
  doi_url TEXT,
  ano VARCHAR(4),
  criado_em TIMESTAMP DEFAULT now(),
  atualizado_em TIMESTAMP DEFAULT now()
);

CREATE TABLE tboportunidades (
  id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  titulo TEXT NOT NULL,
  descricao TEXT,
  requisitos TEXT,
  processo_selecao TEXT,
  prazo TEXT,
  contato TEXT,
  data_criacao DATE DEFAULT CURRENT_DATE
);
